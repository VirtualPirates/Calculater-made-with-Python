Ch = input("enter your choice")
A = int(input("enter no."))
B = int(input("enter another no."))
if Ch =="+":
    print(A+B)
elif Ch =="-":
    print(A-B)
elif Ch=="รท":
    print(A/B)
elif Ch=="*":
    print(A*B)
else:
    print("wrong operator")
